use sr;
SELECT u.iduser, first_name, last_name, email, active, r.description 
                FROM user as u 
                JOIN user_has_role as ur ON ur.iduser = u.iduser
                JOIN role as r ON r.idrole = ur.idrole 
                ORDER BY last_name;
